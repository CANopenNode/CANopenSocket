155 static inline bool_t CO_fifo_putc(CO_fifo_t *fifo, const char c) {
156  if (fifo != NULL && fifo->buf != NULL) {
157  size_t writePtrNext = fifo->writePtr + 1;
158  if (writePtrNext != fifo->readPtr &&
159  !(writePtrNext == fifo->bufSize && fifo->readPtr == 0))
160  {
161  fifo->buf[fifo->writePtr] = c;
162  fifo->writePtr = (writePtrNext == fifo->bufSize) ? 0 : writePtrNext;
163  return true;
164  }
165  }
166  return false;
167 }
