172 static inline bool_t CO_fifo_putc(CO_fifo_t *fifo, const char c) {
173  if (fifo != NULL && fifo->buf != NULL) {
174  size_t writePtrNext = fifo->writePtr + 1;
175  if (writePtrNext != fifo->readPtr &&
176  !(writePtrNext == fifo->bufSize && fifo->readPtr == 0))
177  {
178  fifo->buf[fifo->writePtr] = c;
179  fifo->writePtr = (writePtrNext == fifo->bufSize) ? 0 : writePtrNext;
180  return true;
181  }
182  }
183  return false;
184 }
